```mermaid
graph TD
    subgraph Пользователь / Клиент
        Start([Запрос на товар]) --> MakeOrder[Оформить заказ онлайн]
        ReceiveNotice[Получение товара] --> RateService[Оценка сервиса]
    end

    subgraph Менеджер интернет-магазина
        GetOrder[Получение заказа] --> CreateTask[Создать заявку для склада]
        IssueGoods[Выдача товара клиенту] --> CloseTask[Закрыть заявку в системе]
        CloseTask --> End([Процесс завершен])
    end

    subgraph Склад
        ReceiveTask[Получение заявки в системе] --> CheckStock[Проверить наличие товара]
        CheckStock --> StockCond{Товар в наличии?}
        CallLogist[Вызвать логистику]
        SendToManager[Передать товар менеджеру]
    end

    subgraph Логистика
        Delivery[Доставка товара на склад]
    end

    subgraph Поставщик
        WaitSupplier((Таймер ожидания)) --> DeliverSupplier[Поставщик доставляет товар]
    end

    %% Межролевые связи (сообщения и переходы)
    MakeOrder -.->|Запрос| GetOrder
    CreateTask -.->|Заявка| ReceiveTask
    
    StockCond -- Да --> CallLogist
    CallLogist --> SendToManager
    
    StockCond -- Нет --> WaitSupplier
    WaitSupplier --> DeliverSupplier
    DeliverSupplier --> Delivery
    Delivery --> SendToManager
    
    SendToManager -.->|Готовность| IssueGoods
    SendToManager -.->|Уведомление SMS/Email| ReceiveNotice
    ReceiveNotice --> IssueGoods
    
    IssueGoods --> End
